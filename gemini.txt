# 🧠 Segundo Cerebro Legislativo - Master Blueprint
> **Versión**: 1.0.0 (Electron Embebido + React 18 + SQLite)
> **Objetivo**: Aplicación de escritorio local para gestión legislativa, con UI perfeccionada, control de quórum, versión de proyectos, QR, bot de Telegram y sincronización con GitHub.

---

## 🏗️ Arquitectura & Stack
| Capa | Tecnología |
|------|------------|
| **Frontend (UI)** | React 18, TailwindCSS, Lucide-React |
| **Build Tool** | Vite 5 |
| **Desktop** | Electron 30 (Main Process) |
| **Database** | SQLite (via `better-sqlite3`, archivo local) |
| **Extras** | Telegraf (Bot), Simple-Git (Sync), Winston (Logs), QRCode |

// 📁 Rutas
const DB_PATH = path.join(app.getPath('userData'), 'scl_local.db');
const LOG_PATH = path.join(app.getPath('userData'), 'logs');
const fs = require('fs');
if (!fs.existsSync(LOG_PATH)) fs.mkdirSync(LOG_PATH, { recursive: true });

const db = new Database(DB_PATH);

// 🪵 Logger
const logger = winston.createLogger({
  level: 'info',
  format: winston.format.json(),
  transports: [new winston.transports.File({ filename: path.join(LOG_PATH, 'app.log') })]
});

// 🗄️ DB Init
function initDB() {
  db.exec(`
    CREATE TABLE IF NOT EXISTS audit_logs (id INTEGER PRIMARY KEY, timestamp TEXT, level TEXT, action TEXT, entity TEXT, signature TEXT);
    CREATE TABLE IF NOT EXISTS laws (id INTEGER PRIMARY KEY, nombre TEXT, gaceta TEXT, tipo TEXT, anio INTEGER, ruta TEXT, qr_data TEXT, descargas INTEGER DEFAULT 0);
    CREATE TABLE IF NOT EXISTS attendance (id INTEGER PRIMARY KEY, session_id INTEGER, legislator_id INTEGER, timestamp TEXT);
  `);
}

// 🤖 Telegram
function initBot() {
  if (!process.env.TELEGRAM_BOT_TOKEN) return;
  try {
    const bot = new Telegraf(process.env.TELEGRAM_BOT_TOKEN);
    bot.start((ctx) => ctx.reply('✅ Segundo Cerebro conectado.'));
    bot.command('help', (ctx) => ctx.reply('Comandos: /status, /sync'));
    bot.launch();
    logger.info('Telegram bot launched');
    ipcMain.handle('tg:notify', async (_, payload) => {
      try {
        await bot.telegram.sendMessage(payload.chatId, payload.message);
        return true;
      } catch (e) { return false; }
    });
  } catch (e) { logger.error('Telegram bot failed', e.message); }
}

ipcMain.handle('git:sync', async (_, msg) => {
  try {
    const git = simpleGit();
    await git.add('.').commit(msg || 'auto-sync [electron]');
    await git.push();
    return 'synced';
  } catch (e) { return 'error: ' + e.message; }
});

// Ejemplo de handler DB
ipcMain.handle('db:insert', (_, { table, data }) => {
  const cols = Object.keys(data).join(', ');
  const vals = Object.values(data).map(v => typeof v === 'string' ? `'${v}'` : v).join(', ');
  db.run(`INSERT INTO ${table} (${cols}) VALUES (${vals})`);
  return true;
});

const getDaysSince = (dateStr) => {
  if (!dateStr) return 999;
  const date = new Date(dateStr);
  const now = new Date();
  const diff = now.getTime() - date.getTime();
  return Math.floor(diff / (1000 * 60 * 60 * 24));
};

const getStagnationColor = (dateStr) => {
  const days = getDaysSince(dateStr);
  if (days > 30) return 'text-red-500 bg-red-500/10 border-red-500/20';
  if (days >= 15) return 'text-amber-500 bg-amber-500/10 border-amber-500/20';
  return 'text-emerald-500 bg-emerald-500/10 border-emerald-500/20';
};

const getStagnationLabel = (dateStr) => {
  const days = getDaysSince(dateStr);
  if (days > 30) return 'Estancado (>30 días)';
  if (days >= 15) return 'Atención (15-30 días)';
  return 'Activo (<15 días)';
};

const getSessionTypeByDate = (dateStr) => {
  if (!dateStr) return 'Ordinaria';
  const date = new Date(dateStr + 'T12:00:00');
  const day = date.getDay();
  if (day === 2 || day === 4) return 'Ordinaria';
  return 'Extraordinaria';
};

const generateSessionNumber = (tipo, year, existingSessions) => {
  if (tipo === 'Especial' || tipo === 'Solemne' || tipo === 'Instalación') return '';
  const yearSessions = existingSessions.filter(s => 
    s.activo &&
    (s.tipo === 'Ordinaria' || s.tipo === 'Extraordinaria') &&
    s.tipo === tipo &&
    s.numero_correlativo &&
    s.numero_correlativo.endsWith(`-${year}`)
  );
  const maxNum = yearSessions.reduce((max, s) => {
    const parts = s.numero_correlativo.split('-');
    const num = parseInt(parts[0], 10);
    return num > max ? num : max;
  }, 0);
  return `${String(maxNum + 1).padStart(3, '0')}-${year}`;
};

          <button onClick={() => step < 2 ? setStep(s => s + 1) : handleComplete()} className="flex-1 py-2.5 rounded-xl bg-indigo-600 hover:bg-indigo-700 text-white font-medium transition-all">{step < 2 ? 'Siguiente' : 'Comenzar'}</button>

// COMMAND PALETTE
// ============================================================
const CommandPalette = ({ isOpen, onClose, sessions, oficios, projects, legislators, onNavigate }) => {
  const [query, setQuery] = useState('');
  const inputRef = useRef(null);

  useEffect(() => { if (isOpen && inputRef.current) inputRef.current.focus(); setQuery(''); }, [isOpen]);
  useEffect(() => {
    if (!isOpen) return;
    const h = (e) => { if (e.key === 'Escape') onClose(); };
    window.addEventListener('keydown', h);
    return () => window.removeEventListener('keydown', h);
  }, [isOpen, onClose]);

  const results = useMemo(() => {
    if (!query.trim()) return [];
    const q = query.toLowerCase();
    const found = [];
    sessions?.filter(s => s.activo).forEach(s => { if ((s.numero_correlativo||'').toLowerCase().includes(q) || s.tipo?.toLowerCase().includes(q) || s.fecha?.includes(q)) found.push({ type: 'sesion', id: s.id, label: `${s.numero_correlativo || s.tipo} - ${s.fecha}` }); });
    oficios?.filter(o => o.activo).forEach(o => { if (o.numero_oficio?.toLowerCase().includes(q) || o.asunto?.toLowerCase().includes(q)) found.push({ type: 'oficio', id: o.id, label: `${o.numero_oficio} - ${o.asunto}` }); });
    projects?.filter(p => p.activo).forEach(p => { if (p.titulo?.toLowerCase().includes(q)) found.push({ type: 'proyecto', id: p.id, label: p.titulo }); });
    legislators?.filter(l => l.activo).forEach(l => { if (l.nombre?.toLowerCase().includes(q)) found.push({ type: 'legislador', id: l.id, label: l.nombre }); });
    return found.slice(0, 10);
  }, [query, sessions, oficios, projects, legislators]);

  if (!isOpen) return null;
  return (
    <div className="fixed inset-0 z-[9998] flex items-start justify-center pt-[20vh] bg-black/60 backdrop-blur-sm" onClick={onClose}>
      <div className="w-full max-w-2xl bg-white dark:bg-gray-900 rounded-2xl shadow-2xl border border-gray-200 dark:border-gray-700 overflow-hidden" style={{ animation: 'scale-in 0.2s ease-out' }} onClick={e => e.stopPropagation()}>
        <div className="flex items-center gap-3 px-4 py-3 border-b border-gray-200 dark:border-gray-700">
          <Search className="w-5 h-5 text-gray-400" />
          <input ref={inputRef} type="text" placeholder="Buscar sesiones, oficios, proyectos..." className="flex-1 bg-transparent outline-none text-gray-900 dark:text-white placeholder-gray-400" value={query} onChange={e => setQuery(e.target.value)} />
          <kbd className="px-2 py-1 text-xs bg-gray-100 dark:bg-gray-800 rounded-md text-gray-500">ESC</kbd>
        </div>
        <div className="max-h-80 overflow-y-auto p-2">
          {results.length === 0 && query && <p className="text-center py-8 text-gray-400 text-sm">Sin resultados</p>}
          {!query && <div className="py-8 text-center text-gray-400 text-sm"><Search className="w-8 h-8 mx-auto mb-2 opacity-40" /><p>Escribe para buscar en toda la aplicación</p></div>}
          {results.map((r, i) => (
            <div key={i} className="w-full flex items-center gap-3 px-3 py-2.5 rounded-lg hover:bg-gray-100 dark:hover:bg-gray-800 text-left transition-colors cursor-pointer" onClick={() => { if (onNavigate && r.type && r.id) onNavigate(r.type, r.id); onClose(); }}>
              <span className="text-xs font-medium px-2 py-0.5 rounded-full bg-blue-100 dark:bg-blue-900/30 text-blue-600 dark:text-blue-400">{r.type}</span>
              <span className="text-sm text-gray-900 dark:text-gray-100 truncate">{r.label}</span>
            </div>
          ))}
        </div>
      </div>
    </div>
  );
};

const SessionsModule = ({ sessions, oficios, onSave, onDelete, darkMode, addToast, onNavigate }) => {
  const [view, setView] = useState('list');
  const [editingId, setEditingId] = useState(null);
  const [form, setForm] = useState({ tipo: 'Ordinaria', numero_correlativo: '', motivo: '', fecha: new Date().toISOString().split('T')[0], hora_inicio: '09:00', hora_cierre: '12:00', periodo: '2026-2027', observaciones: '' });
  const resetForm = () => { setForm({ tipo: 'Ordinaria', numero_correlativo: '', motivo: '', fecha: new Date().toISOString().split('T')[0], hora_inicio: '09:00', hora_cierre: '12:00', periodo: '2026-2027', observaciones: '' }); setEditingId(null); setView('list'); };

  const handleDateChange = (val) => { const tipo = getSessionTypeByDate(val); const year = new Date(val + 'T12:00:00').getFullYear(); setForm(prev => ({ ...prev, fecha: val, tipo, numero_correlativo: generateSessionNumber(tipo, year, sessions) })); };
  const handleTypeChange = (val) => { const needsNumber = val === 'Ordinaria' || val === 'Extraordinaria'; setForm(prev => { if (needsNumber && prev.fecha) { const year = new Date(prev.fecha + 'T12:00:00').getFullYear(); return { ...prev, tipo: val, numero_correlativo: generateSessionNumber(val, year, sessions), motivo: '' }; } return { ...prev, tipo: val, numero_correlativo: '', motivo: '' }; }); };

  const handleSave = () => {
    if (!form.fecha) { addToast('La fecha es requerida', 'error'); return; }
    if ((form.tipo === 'Especial' || form.tipo === 'Solemne' || form.tipo === 'Instalación') && !form.motivo) { addToast('El motivo es requerido', 'error'); return; }
    const isDuplicate = sessions.some(s => s.activo && s.id !== editingId && s.numero_correlativo === form.numero_correlativo && s.tipo === form.tipo && form.numero_correlativo);
    if (isDuplicate) { addToast('Ya existe una sesión con este número correlativo', 'error'); return; }
    onSave(editingId ? { ...form, id: editingId } : form);
    addToast(editingId ? 'Sesión actualizada' : 'Sesión creada', 'success'); resetForm();
  };

  const handleDelete = (id) => {
    const linked = oficios.filter(o => o.activo && o.sesion_id === id);
    if (linked.length > 0 && !window.confirm(`⚠️ Esta sesión tiene ${linked.length} oficio(s) vinculado(s).\n\n¿Deseas eliminarla igualmente?`)) return;
    onDelete(id); addToast('Sesión eliminada', 'warning');
  };

  const getLinkedOficios = useCallback((sesionId) => oficios.filter(o => o.activo && o.sesion_id === sesionId), [oficios]);
  const typeColors = { 'Ordinaria': 'bg-blue-100 text-blue-700 dark:bg-blue-500/10 dark:text-blue-400', 'Extraordinaria': 'bg-purple-100 text-purple-700 dark:bg-purple-500/10 dark:text-purple-400', 'Especial': 'bg-amber-100 text-amber-700 dark:bg-amber-500/10 dark:text-amber-400', 'Solemne': 'bg-rose-100 text-rose-700 dark:bg-rose-500/10 dark:text-rose-400', 'Instalación': 'bg-cyan-100 text-cyan-700 dark:bg-cyan-500/10 dark:text-cyan-400' };
  const activeSessions = useMemo(() => sessions.filter(s => s.activo).sort((a, b) => b.fecha.localeCompare(a.fecha)), [sessions]);
  const inputCls = `w-full px-4 py-2.5 rounded-xl border outline-none focus:ring-2 focus:ring-indigo-500 ${darkMode ? 'bg-gray-800 border-gray-700 text-white' : 'bg-gray-50 border-gray-200'}`;
  const labelCls = `block text-sm mb-1 ${darkMode ? 'text-gray-400' : 'text-gray-600'}`;

  if (view === 'form' || view === 'edit') return (
    <div className="max-w-3xl mx-auto space-y-6">
      <button onClick={resetForm} className={`flex items-center gap-2 text-sm ${darkMode ? 'text-gray-400 hover:text-white' : 'text-gray-500 hover:text-gray-900'} transition-colors`}><ChevronLeft className="w-4 h-4" /> Volver al listado</button>
      <div className={`rounded-2xl border p-6 ${darkMode ? 'bg-gray-900 border-gray-800' : 'bg-white border-gray-200'}`}>
        <h2 className="text-xl font-semibold mb-6">{editingId ? 'Editar Sesión' : 'Nueva Sesión'}</h2>
        <div className="space-y-4">
          <div><label className={labelCls}>Fecha</label><input type="date" value={form.fecha} onChange={e => handleDateChange(e.target.value)} className={inputCls} />{form.fecha && <p className={`text-xs mt-1 ${darkMode ? 'text-gray-500' : 'text-gray-400'}`}>Tipo detectado: <span className="font-medium">{form.tipo}</span></p>}</div>
          <div><label className={labelCls}>Tipo de Sesión</label><select value={form.tipo} onChange={e => handleTypeChange(e.target.value)} className={inputCls}><option value="Ordinaria">Ordinaria</option><option value="Extraordinaria">Extraordinaria</option><option value="Especial">Especial</option><option value="Solemne">Solemne</option><option value="Instalación">Instalación</option></select></div>
          {(form.tipo === 'Ordinaria' || form.tipo === 'Extraordinaria') && (<div><label className={labelCls}>Número Correlativo</label><div className="flex gap-2"><input type="text" value={form.numero_correlativo} readOnly className={`flex-1 px-4 py-2.5 rounded-xl border ${darkMode ? 'bg-gray-800 border-gray-700 text-gray-400' : 'bg-gray-100 border-gray-200 text-gray-500'}`} /><button onClick={() => handleTypeChange(form.tipo)} className={`px-4 py-2.5 rounded-xl border ${darkMode ? 'border-gray-700 hover:bg-gray-800' : 'border-gray-200 hover:bg-gray-50'}`}><RefreshCw className="w-4 h-4" /></button></div></div>)}
          {(form.tipo === 'Especial' || form.tipo === 'Solemne' || form.tipo === 'Instalación') && (<div><label className={labelCls}>Motivo / Período</label><input type="text" value={form.motivo} onChange={e => setForm(prev => ({ ...prev, motivo: e.target.value }))} placeholder="Ej: Conmemoración bicentenario" className={`${inputCls} ${darkMode ? 'placeholder-gray-500' : ''}`} /></div>)}
          <div className="grid grid-cols-2 gap-4"><div><label className={labelCls}>Hora Inicio</label><input type="time" value={form.hora_inicio} onChange={e => setForm(p => ({ ...p, hora_inicio: e.target.value }))} className={inputCls} /></div><div><label className={labelCls}>Hora Cierre</label><input type="time" value={form.hora_cierre} onChange={e => setForm(p => ({ ...p, hora_cierre: e.target.value }))} className={inputCls} /></div></div>

'Actualizar' : 'Crear Sesión'}</button></div>
      </div>
    </div>
  );

  return (
    <div className="space-y-6">
      <div className="flex items-center justify-between"><div><h1 className="text-2xl font-bold mb-1">Sesiones</h1><p className={`text-sm ${darkMode ? 'text-gray-400' : 'text-gray-500'}`}>Gestión de sesiones legislativas</p></div><button onClick={() => { setView('form'); setEditingId(null); setForm({ tipo: 'Ordinaria', numero_correlativo: '', motivo: '', fecha: new Date().toISOString().split('T')[0], hora_inicio: '09:00', hora_cierre: '12:00', periodo: '2026-2027', observaciones: '' }); }} className="flex items-center gap-2 px-4 py-2.5 bg-indigo-600 hover:bg-indigo-700 text-white rounded-xl text-sm font-medium"><Plus className="w-4 h-4" /> Nueva Sesión</button></div>
      <div className={`rounded-2xl border overflow-hidden ${darkMode ? 'bg-gray-900 border-gray-800' : 'bg-white border-gray-200'}`}>
        <div className="overflow-x-auto">
          <table className="w-full"><thead><tr className={`border-b ${darkMode ? 'border-gray-800' : 'border-gray-100'}`}><th className={`text-left px-6 py-3 text-xs font-medium uppercase ${darkMode ? 'text-gray-500' : 'text-gray-400'}`}>#</th><th className={`text-left px-6 py-3 text-xs font-medium uppercase ${darkMode ? 'text-gray-500' : 'text-gray-400'}`}>Tipo</th><th className={`text-left px-6 py-3 text-xs font-medium uppercase ${darkMode ? 'text-gray-500' : 'text-gray-400'}`}>Fecha</th><th className={`text-left px-6 py-3 text-xs font-medium uppercase ${darkMode ? 'text-gray-500' : 'text-gray-400'}`}>Horario</th><th className={`text-left px-6 py-3 text-xs font-medium uppercase ${darkMode ? 'text-gray-500' : 'text-gray-400'}`}>Oficios</th><th className={`text-right px-6 py-3 text-xs font-medium uppercase ${darkMode ? 'text-gray-500' : 'text-gray-400'}`}>Acciones</th></tr></thead>
          <tbody>{activeSessions.map(s => { const ofs = getLinkedOficios(s.id); return (<tr key={s.id} className={`border-b last:border-0 transition-colors ${darkMode ? (<button onClick={(e) => { e.stopPropagation(); onNavigate?.('oficio', ofs[0].id); }} className="inline-flex items-center gap-1 text-xs text-blue-500 hover:text-blue-600 transition-colors" title="Ver oficio vinculado"><Link className="w-3 h-3" /> {ofs.length}</button>) : (<span className={`text-xs ${darkMode ? 'text-gray-600' : 'text-gray-300'}`}>—</span>)}</td><td className="px-6 py-4"><div className="flex items-center justify-end gap-1"><button onClick={() => { setEditingId(s.id); setForm({ tipo: s.tipo, numero_correlativo: s.numero_correlativo, motivo: s.motivo, fecha: s.fecha, hora_inicio: s.hora_inicio, hora_cierre: s.hora_cierre, periodo: s.periodo, observaciones: s.observaciones }); setView('edit'); }} className={`p-2 rounded-lg ${darkMode ? 'text-gray-700' : 'text-gray-300'}`} /><p className={`text-sm ${darkMode ? 'text-gray-500' : 'text-gray-400'}`}>No hay sesiones registradas</p></td></tr>)}</tbody></table>

const VaultModule = ({ documents, sessions, oficios, projects, darkMode, addToast, onSaveDocument, onDeleteDocument }) => {
  const [filterType, setFilterType] = useState('');
  const [showUpload, setShowUpload] = useState(false);
  const [uploadForm, setUploadForm] = useState({ entidad_tipo: 'Sesion', entidad_id: '', fase_etiqueta: '', nombre_archivo: '' });
  const [selectedFile, setSelectedFile] = useState(null);
  const [uploadProgress, setUploadProgress] = useState(0);

  const getEntityName = (doc) => {
    if (doc.entidad_tipo === 'Sesion') { const s = sessions.find(s => s.id === doc.entidad_id); return s ? `${s.numero_correlativo || s.tipo} (${s.fecha})` : 'No encontrada'; }
    if (doc.entidad_tipo === 'Oficio') { const o = oficios.find(o => o.id === doc.entidad_id); return o ? o.numero_oficio : 'No encontrado'; }
    if (doc.entidad_tipo === 'ProyectoLey') { const p = projects.find(p => p.id === doc.entidad_id); return p ? p.titulo : 'No encontrado'; }
    return '';
  };
  const getPhaseLabels = (t) => { if (t === 'Sesion') return ['Acta', 'Acuerdo', 'Orden del Día', 'Asistencia']; if (t === 'Oficio') return ['Oficio Original', 'Acuse de Recibo', 'Respuesta']; if (t === 'ProyectoLey') return ['Estudio en Comisión', 'Informe de Dirección', '1ra Discusión', '2da Discusión', '3ra Discusión', 'Aprobada', 'Gaceta Oficial']; return []; };

  const handleFileSelect = (e) => { const file = e.target.files?.[0]; if (!file) return; if (!['application/pdf', 'application/msword', 'application/vnd.openxmlformats-officedocument.wordprocessingml.document'].includes(file.type)) { addToast('Solo PDF o Word', 'error'); return; } if (file.size > 10 * 1024 * 1024) { addToast('Máx. 10MB', 'error'); return; } setSelectedFile(file); setUploadForm(p => ({ ...p, nombre_archivo: file.name })); };

  const handleUpload = async () => {
    if (!uploadForm.entidad_id) { addToast('Seleccione una entidad', 'error'); return; }
    if (!uploadForm.fase_etiqueta) { addToast('Seleccione la fase', 'error'); return; }
    try {
      setUploadProgress(30);
      let base64 = null, fileSize = 0, fileType = '';
      if (selectedFile) { base64 = await fileToBase64(selectedFile); fileSize = selectedFile.size; fileType = selectedFile.type; setUploadProgress(70); }
      const hash = `sha256:${Array.from(uploadForm.nombre_archivo || 'doc').reduce((a, c) => a + c.charCodeAt(0), 0).toString(16).padStart(8, '0')}`;
      onSaveDocument?.({ id: Date.now(), entidad_tipo: uploadForm.entidad_tipo, entidad_id: Number(uploadForm.entidad_id), fase_etiqueta: uploadForm.fase_etiqueta, ruta_archivo: `/boveda/${uploadForm.entidad_tipo.toLowerCase()}/${uploadForm.entidad_id}/${uploadForm.nombre_archivo || 'doc'}`, hash_integridad: hash, nombre_original: uploadForm.nombre_archivo || 'documento', tipo_mime: fileType || 'application/octet-stream', tamano_bytes: fileSize, fecha_subida: new Date().toISOString().split('T')[0], contenido_base64: base64, activo: true });
      setUploadProgress(100); addToast(`✅ "${uploadForm.nombre_archivo}" subido`, 'success');
      setShowUpload(false); setSelectedFile(null); setUploadProgress(0); setUploadForm({ entidad_tipo: 'Sesion', entidad_id: '', fase_etiqueta: '', nombre_archivo: '' });
    } catch { addToast('Error al subir', 'error'); setUploadProgress(0); }
  };

  const activeDocs = useMemo(() => documents.filter(d => { if (!d.activo) return false; if (filterType && d.entidad_tipo !== filterType) return false; return true; }), [documents, filterType]);
  const activeSessions = useMemo(() => sessions.filter(s => s.activo), [sessions]);
  const activeOficios = useMemo(() => oficios.filter(o => o.activo), [oficios]);
  const activeProjects = useMemo(() => projects.filter(p => p.activo), [projects]);
  const typeIcons = { Sesion: <CalendarDays className="w-5 h-5" />, Oficio: <FileText className="w-5 h-5" />, ProyectoLey: <Scale className="w-5 h-5" /> };
  const typeColors = { Sesion: 'text-blue-500 bg-blue-500/10', Oficio: 'text-purple-500 bg-purple-500/10', ProyectoLey: 'text-emerald-500 bg-emerald-500/10' };
  const selectCls = `w-full px-4 py-2.5 rounded-xl border outline-none focus:ring-2 focus:ring-indigo-500 ${darkMode ? 'bg-gray-800 border-gray-700 text-white' : 'bg-gray-50 border-gray-200'}`;
  const labelCls = `block text-sm mb-1 ${darkMode ? 'text-gray-400' : 'text-gray-600'}`;

  return (
    <div className="space-y-6">
      <div className="flex items-center justify-between"><div><h1 className="text-2xl font-bold mb-1">Bóveda Documental</h1><p className={`text-sm ${darkMode ? 'text-gray-400' : 'text-gray-500'}`}>Gestión y almacenamiento de documentos</p></div><button onClick={() => setShowUpload(true)} className="flex items-center gap-2 px-4 py-2.5 bg-indigo-600 hover:bg-indigo-700 text-white rounded-xl text-sm font-medium"><Upload className="w-4 h-4" /> Subir Archivo</button></div>
      <div className="flex gap-1 p-1 rounded-xl bg-gray-100 dark:bg-gray-800 w-fit">
        {['Todos', 'Sesion', 'Oficio', 'ProyectoLey'].map((t) => (<button key={t} onClick={() => setFilterType(t === 'Todos' ? '' : t)} className={`px-4 py-2 rounded-lg text-sm font-medium ${(!filterType && t === 'Todos') || filterType === t ? 'bg-white dark:bg-gray-700 shadow-sm' : ''}`}>{t === 'Sesion' ? 'Sesiones' : t === 'ProyectoLey' ? 'Leyes' : t}</button>))}
      </div>
      <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4">
        {activeDocs.map(doc => (
          <div key={doc.id} className={`rounded-2xl border p-5 ${darkMode ? 'bg-gray-900 border-gray-800' : 'bg-white border-gray-200'}`}>
            <div className="flex items-start gap-3"><div className={`p-2.5 rounded-xl ${typeColors[doc.entidad_tipo] || ''}`}>{typeIcons[doc.entidad_tipo] || <File className="w-5 h-5" />}</div><div className="min-w-0 flex-1"><p className="text-sm font-medium truncate">{getEntityName(doc)}</p><p className={`text-xs mt-0.5 ${darkMode ? 'text-gray-500' : 'text-gray-400'}`}>{doc.entidad_tipo}</p></div></div>
            <div className={`mt-4 p-3 rounded-xl ${darkMode ? 'bg-gray-800' : 'bg-gray-50'}`}><p className={`text-xs ${darkMode ? 'text-gray-500' : 'text-gray-400'}`}>Fase</p><p className="text-sm font-medium">{doc.fase_etiqueta}</p></div>
            <div className={`mt-2 p-3 rounded-xl ${darkMode ? 'bg-gray-800' : 'bg-gray-50'}`}><p className={`text-xs ${darkMode ? 'text-gray-500' : 'text-gray-400'}`}>Ruta</p><p className={`text-xs font-mono truncate ${darkMode ? 'text-gray-400' : 'text-gray-500'}`}>{doc.ruta_archivo}</p></div>
            <div className="flex items-center justify-between mt-3">
              <span className={`text-[10px] font-mono ${darkMode ? 'text-gray-600' : 'text-gray-300'}`}>{doc.tipo_mime?.split('/')?.[1]?.toUpperCase() || 'DOC'} • {doc.tamano_bytes ? `${(doc.tamano_bytes / 1024).toFixed(1)} KB` : '—'}</span>
              <div className="flex gap-1">
                {doc.contenido_base64 && (<a href={doc.contenido_base64} download={doc.nombre_original} className={`p-1.5 rounded-lg ${darkMode ? 'hover:bg-gray-800' : 'hover:bg-gray-100'}`} title="Descargar"><svg width="14" height="14" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2"><path d="M21 15v4a2 2 0 0 1-2 2H5a2 2 0 0 1-2-2v-4"/><polyline points="7 10 12 15 17 10"/><line x1="12" y1="15" x2="12" y2="3"/></svg></a>)}
                <button className={`p-1.5 rounded-lg ${darkMode ? 'hover:bg-gray-800' : 'hover:bg-gray-100'}`} title="Detalles" onClick={(e) => { e.stopPropagation(); addToast('Detalles del documento', 'info'); }}><Info className="w-3.5 h-3.5" /></button>
                <button className="p-1.5 rounded-lg text-red-500 hover:bg-red-500/10" title="Eliminar" onClick={(e) => { e.stopPropagation(); if (window.confirm('¿Eliminar?')) { onDeleteDocument?.(doc.id); addToast('Eliminado', 'warning'); } }}><Trash2 className="w-3.5 h-3.5" /></button>
              </div>
            </div>
          </div>
        ))}
        {activeDocs.length === 0 && (<div className={`col-span-full rounded-2xl border p-12 text-center ${darkMode ? 'bg-gray-900 border-gray-800' : 'bg-white border-gray-200'}`}><FolderOpen className={`w-12 h-12 mx-auto mb-3 ${darkMode ? 'text-gray-700' : 'text-gray-300'}`} /><p className={`text-sm ${darkMode ? 'text-gray-500' : 'text-gray-400'}`}>No hay documentos</p></div>)}
      </div>

      <Modal isOpen={showUpload} onClose={() => { setShowUpload(false); setSelectedFile(null); setUploadProgress(0); }} title="Subir Archivo" darkMode={darkMode}>
        <div className="space-y-4">
          <div className={`p-6 rounded-xl border-2 border-dashed text-center ${darkMode ? 'border-gray-700 bg-gray-800 hover:border-indigo-500' : 'border-gray-200 bg-gray-50 hover:border-indigo-400'}`}>
            <input type="file" id="file-upload" className="hidden" accept=".pdf,.doc,.docx" onChange={handleFileSelect} />
            <label htmlFor="file-upload" className="cursor-pointer flex flex-col items-center">
              <Upload className={`w-10 h-10 mx-auto mb-3 ${selectedFile ? 'text-emerald-500' : darkMode ? 'text-gray-600' : 'text-gray-300'}`} />
              {selectedFile ? (<><p className="text-sm font-medium text-emerald-500 truncate max-w-xs">{selectedFile.name}</p><p className={`text-xs ${darkMode ? 'text-gray-500' : 'text-gray-400'}`}>{(selectedFile.size / 1024 / 1024).toFixed(2)} MB</p><button type="button" onClick={e => { e.preventDefault(); setSelectedFile(null); }} className="mt-2 text-xs text-red-500 hover:underline">Remover</button></>) : (<><p className={`text-sm ${darkMode ? 'text-gray-400' : 'text-gray-500'}`}>Arrastra o <span className="text-indigo-500 underline">explora</span></p><p className={`text-xs mt-1 ${darkMode ? 'text-gray-600' : 'text-gray-300'}`}>PDF, DOC, DOCX (máx. 10MB)</p></>)}
            </label>
            {uploadProgress > 0 && uploadProgress < 100 && (<div className="mt-4 w-full bg-gray-200 dark:bg-gray-700 rounded-full h-1.5"><div className="bg-indigo-500 h-1.5 rounded-full transition-all" style={{ width: `${uploadProgress}%` }} /></div>)}
          </div>
          <div><label className={labelCls}>Tipo de Entidad</label><select value={uploadForm.entidad_tipo} onChange={e => setUploadForm(p => ({ ...p, entidad_tipo: e.target.value, entidad_id: '', fase_etiqueta: '' }))} className={selectCls}><option value="Sesion">Sesión</option><option value="Oficio">Oficio</option><option value="ProyectoLey">Proyecto de Ley</option></select></div>
          <div><label className={labelCls}>Entidad</label><select value={uploadForm.entidad_id} onChange={e => setUploadForm(p => ({ ...p, entidad_id: e.target.value }))} className={selectCls}><option value="">Seleccionar...</option>{(uploadForm.entidad_tipo === 'Sesion' ? activeSessions : uploadForm.entidad_tipo === 'Oficio' ? activeOficios : activeProjects).map(e => (<option key={e.id} value={e.id}>{uploadForm.entidad_tipo === 'Sesion' ? (e.numero_correlativo || e.tipo) : uploadForm.entidad_tipo === 'Oficio' ? e.numero_oficio : e.titulo}</option>))}</select></div>
          <div><label className={labelCls}>Fase</label><select value={uploadForm.fase_etiqueta} onChange={e => setUploadForm(p => ({ ...p, fase_etiqueta: e.target.value }))} className={selectCls}><option value="">Seleccionar...</option>{getPhaseLabels(uploadForm.entidad_tipo).map(f => <option key={f} value={f}>{f}</option>)}</select></div>
          <button onClick={handleUpload} disabled={uploadProgress > 0 && uploadProgress < 100} className="w-full py-2.5 rounded-xl bg-indigo-600 hover:bg-indigo-700 disabled:opacity-50 text-white font-medium flex items-center justify-center gap-2">{uploadProgress === 100 ? <Check className="w-4 h-4" /> : <Upload className="w-4 h-4" />}{uploadProgress === 100 ? '¡Subido!' : 'Subir'}</button>
        </div>
      </Modal>
    </div>
  );
};

const AgendaModule = ({ projects, commissions, legislators, onSave, onDelete, darkMode, addToast, onNavigate }) => {
  const [view, setView] = useState('list');
  const [editingId, setEditingId] = useState(null);
  const [form, setForm] = useState({ titulo: '', origen: 'Comisión', comision_id: '', ponente_id: '', fase_actual: 'Estudio en Comisión', urgencia_parlamentaria: false, fecha_ingreso: new Date().toISOString().split('T')[0] });
  const [filters, setFilters] = useState({ comision_id: '', origen: '', fase: '', urgencia: '' });
  const [detailProject, setDetailProject] = useState(null);
  const resetForm = () => { setForm({ titulo: '', origen: 'Comisión', comision_id: '', ponente_id: '', fase_actual: 'Estudio en Comisión', urgencia_parlamentaria: false, fecha_ingreso: new Date().toISOString().split('T')[0] }); setEditingId(null); setView('list'); };
  const handleOriginChange = (val) => { setForm(prev => ({ ...prev, origen: val, fase_actual: getRoutePhases(val)[0] })); };
  const handleSave = () => { if (!form.titulo.trim()) { addToast('Título requerido', 'error'); return; } const now = new Date().toISOString().split('T')[0]; onSave(editingId ? { ...form, id: editingId, comision_id: form.comision_id ? Number(form.comision_id) : null, ponente_id: form.ponente_id ? Number(form.ponente_id) : null, fecha_actualizacion: now } : { ...form, comision_id: form.comision_id ? Number(form.comision_id) : null, ponente_id: form.ponente_id ? Number(form.ponente_id) : null, fecha_actualizacion: now }); addToast(editingId ? 'Actualizado' : 'Creado', 'success'); resetForm(); };
  const handleAdvancePhase = (project) => {
    if (project.fase_actual === 'Sancionada') { addToast('Ya está sancionado', 'info'); return; }
    const phases = getRoutePhases(project.origen);
    const idx = phases.indexOf(project.fase_actual);
    if (idx >= phases.length - 1) { addToast('Fase final alcanzada', 'info'); return; }
    const next = phases[idx + 1];
    if (!project.urgencia_parlamentaria && next.includes('Discusión') && !project.fase_actual.includes('Discusión')) { if (!confirm(`⚠️ Saltando fases.\nDe "${project.fase_actual}" a "${next}".\n¿Continuar?`)) return; }
    const updated = { ...project, fase_actual: next, fecha_actualizacion: new Date().toISOString().split('T')[0] };
    onSave(updated); setDetailProject(updated); addToast(`Fase: ${next}`, 'success');
  };

  const filteredProjects = useMemo(() => projects.filter(p => { if (!p.activo) return false; if (filters.comision_id && p.comision_id !== Number(filters.comision_id)) return false; if (filters.origen && p.origen !== filters.origen) return false; if (filters.fase && p.fase_actual !== filters.fase) return false; if (filters.urgencia === 'true' && !p.urgencia_parlamentaria) return false; if (filters.urgencia === 'false' && p.urgencia_parlamentaria) return false; return true; }), [projects, filters]);
  const activeLegislators = useMemo(() => legislators.filter(l => l.activo), [legislators]);
  const activeCommissions = useMemo(() => commissions.filter(c => c.activo), [commissions]);
  const inputCls = `w-full px-4 py-2.5 rounded-xl border outline-none focus:ring-2 focus:ring-indigo-500 ${darkMode ? 'bg-gray-800 border-gray-700 text-white' : 'bg-gray-50 border-gray-200'}`;
  const labelCls = `block text-sm mb-1 ${darkMode ? 'text-gray-400' : 'text-gray-600'}`;
  const selectCls = inputCls;

  if (detailProject) {
    const phases = getRoutePhases(detailProject.origen);
    const currentIdx = phases.indexOf(detailProject.fase_actual);
    const commission = commissions.find(c => c.id === detailProject.comision_id);
    const ponente = legislators.find(l => l.id === detailProject.ponente_id);
    return (
      <div className="max-w-4xl mx-auto space-y-6">
        <button onClick={() => setDetailProject(null)} className={`flex items-center gap-2 text-sm ${darkMode ? 'text-gray-400 hover:text-white' : 'text-gray-500 hover:text-gray-900'}`}><ChevronLeft className="w-4 h-4" /> Volver</button>
        <div className={`rounded-2xl border p-6 ${darkMode ? 'bg-gray-900 border-gray-800' : 'bg-white border-gray-200'}`}>
          <div className="flex items-start justify-between mb-6"><div><h2 className="text-xl font-bold">{detailProject.titulo}</h2><div className="flex items-center gap-2 mt-1"><span className={`inline-flex px-2.5 py-1 rounded-lg text-xs font-medium ${detailProject.origen === 'Comisión' ? 'bg-blue-100 text-blue-700 dark:bg-blue-500/10 dark:text-blue-400' : 'bg-purple-100 text-purple-700 dark:bg-purple-500/10 dark:text-purple-400'}`}>{detailProject.origen}</span>{detailProject.urgencia_parlamentaria && <span className="inline-flex items-center gap-1 px-2.5 py-1 rounded-lg text-xs font-medium bg-red-100 text-red-700 dark:bg-red-500/10 dark:text-red-400"><Zap className="w-3 h-3" /> Urgente</span>}</div></div><span className={`px-3 py-1.5 rounded-xl text-xs font-medium border ${getStagnationColor(detailProject.fecha_actualizacion)}`}>{getStagnationLabel(detailProject.fecha_actualizacion)}</span></div>
          <div className="mb-8"><p className={`text-sm mb-4 ${darkMode ? 'text-gray-400' : 'text-gray-500'}`}>Progreso:</p><div className="flex gap-1 overflow-x-auto pb-2">{phases.map((phase, idx) => (<React.Fragment key={phase}><div className={`flex-shrink-0 px-3 py-2 rounded-lg text-xs font-medium whitespace-nowrap ${idx < currentIdx ? 'bg-emerald-100 text-emerald-700 dark:bg-emerald-500/20 dark:text-emerald-400' : ''}${idx === currentIdx ? 'bg-indigo-100 text-indigo-700 dark:bg-indigo-500/20 dark:text-indigo-400 ring-2 ring-indigo-500' : ''}${idx > currentIdx ? darkMode ? 'bg-gray-800 text-gray-500' : 'bg-gray-100 text-gray-400' : ''}`}>{idx < currentIdx && <Check className="w-3 h-3 inline mr-1" />}{phase}</div>{idx < phases.length - 1 && <div className={`flex-shrink-0 w-4 flex items-center justify-center ${idx < currentIdx ? 'text-emerald-500' : darkMode ? 'text-gray-700' : 'text-gray-300'}`}><ArrowRight className="w-3 h-3" /></div>}</React.Fragment>))}</div></div>
          <div className="grid grid-cols-2 md:grid-cols-4 gap-4 mb-6"><div className={`p-3 rounded-xl ${darkMode ? 'bg-gray-800' : 'bg-gray-50'}`}><p className={`text-xs ${darkMode ? 'text-gray-500' : 'text-gray-400'}`}>Comisión</p><p className="text-sm font-medium">{commission?.nombre || '—'}</p></div><div className={`p-3 rounded-xl ${darkMode ? 'bg-gray-800' : 'bg-gray-50'}`}><p className={`text-xs ${darkMode ? 'text-gray-500' : 'text-gray-400'}`}>Ponente</p><p className="text-sm font-medium">{ponente?.nombre || '—'}</p></div><div className={`p-3 rounded-xl ${darkMode ? 'bg-gray-800' : 'bg-gray-50'}`}><p className={`text-xs ${darkMode ? 'text-gray-500' : 'text-gray-400'}`}>Ingreso</p><p className="text-sm font-medium">{detailProject.fecha_ingreso}</p></div><div className={`p-3 rounded-xl ${darkMode ? 'bg-gray-800' : 'bg-gray-50'}`}><p className={`text-xs ${darkMode ? 'text-gray-500' : 'text-gray-400'}`}>Actualización</p><p className="text-sm font-medium">{detailProject.fecha_actualizacion}</p></div></div>
          <div className="flex flex-wrap gap-3">{currentIdx < phases.length - 1 && <button onClick={() => handleAdvancePhase(detailProject)} className="flex-1 min-w-[200px] py-2.5 rounded-xl bg-indigo-600 hover:bg-indigo-700 text-white font-medium flex items-center justify-center gap-2"><ArrowRight className="w-4 h-4" /> Avanzar a {phases[currentIdx + 1]}</button>}<button onClick={() => { setEditingId(detailProject.id); setForm({ titulo: detailProject.titulo, origen: detailProject.origen, comision_id: detailProject.comision_id || '', ponente_id: detailProject.ponente_id || '', fase_actual: detailProject.fase_actual, urgencia_parlamentaria: detailProject.urgencia_parlamentaria, fecha_ingreso: detailProject.fecha_ingreso }); setView('edit'); setDetailProject(null); }} className={`px-4 py-2.5 rounded-xl font-medium flex items-center gap-2 ${darkMode ? 'bg-gray-800 hover:bg-gray-700 text-white' : 'bg-gray-100 hover:bg-gray-200'}`}><Edit3 className="w-4 h-4" /> Editar</button><button onClick={() => { onDelete(detailProject.id); setDetailProject(null); addToast('Eliminado', 'warning'); }} className="px-4 py-2.5 rounded-xl font-medium text-red-500 hover:bg-red-500/10 flex items-center gap-2"><Trash2 className="w-4 h-4" /> Eliminar</button></div>
        </div>
      </div>
    );
  }

  if (view === 'form' || view === 'edit') {

function App() {
  const [config, setConfig] = useState(() => safeJSONParse('scl_config', defaultConfig));
  const [sessions, setSessions] = useState(() => safeJSONParse('scl_sessions', initialSessions));
  const [legislators, setLegislators] = useState(() => safeJSONParse('scl_legislators', initialLegislators));
  const [commissions, setCommissions] = useState(() => safeJSONParse('scl_commissions', initialCommissions));
  const [oficios, setOficios] = useState(() => safeJSONParse('scl_oficios', initialOficios));
  const [projects, setProjects] = useState(() => safeJSONParse('scl_projects', initialProjects));
  const [documents, setDocuments] = useState(() => safeJSONParse('scl_documents', initialDocuments));
  const [sidebarOpen, setSidebarOpen] = useState(true);
  const [currentPage, setCurrentPage] = useState('dashboard');
  const [toasts, setToasts] = useState([]);
  const [showCommandPalette, setShowCommandPalette] = useState(false);
  const darkMode = config.darkMode;

  useEffect(() => { try { localStorage.setItem('scl_config', JSON.stringify(config)); } catch {} }, [config]);
  useEffect(() => { try { localStorage.setItem('scl_sessions', JSON.stringify(sessions)); } catch {} }, [sessions]);
  useEffect(() => { try { localStorage.setItem('scl_legislators', JSON.stringify(legislators)); } catch {} }, [legislators]);
  useEffect(() => { try { localStorage.setItem('scl_commissions', JSON.stringify(commissions)); } catch {} }, [commissions]);
  useEffect(() => { try { localStorage.setItem('scl_oficios', JSON.stringify(oficios)); } catch {} }, [oficios]);
  useEffect(() => { try { localStorage.setItem('scl_projects', JSON.stringify(projects)); } catch {} }, [projects]);
  useEffect(() => { try { localStorage.setItem('scl_documents', JSON.stringify(documents)); } catch {} }, [documents]);

  useEffect(() => { const h = (e) => { if ((e.metaKey || e.ctrlKey) && e.key === 'k') { e.preventDefault(); setShowCommandPalette(p => !p); } }; window.addEventListener('keydown', h); return () => window.removeEventListener('keydown', h); }, []);

  const addToast = useCallback((message, type = 'info') => { const id = Date.now() + Math.random(); setToasts(p => [...p, { id, message, type }]); setTimeout(() => setToasts(p => p.filter(t => t.id !== id)), 4000); }, []);
  const removeToast = useCallback((id) => setToasts(p => p.filter(t => t.id !== id)), []);
  const toggleDarkMode = () => setConfig(p => ({ ...p, darkMode: !p.darkMode }));
  const navigateToEntity = useCallback((type) => {
    const map = { sesion: 'sesiones', oficio: 'oficios', proyecto: 'agenda', legislador: 'legisladores' };
    if (map[type]) setCurrentPage(map[type]);
  }, []);

  const navItems = [
    { id: 'dashboard', label: 'Dashboard', icon: <LayoutDashboard className="w-5 h-5" /> },
    { id: 'sesiones', label: 'Sesiones', icon: <Calendar className="w-5 h-5" /> },
    { id: 'oficios', label: 'Oficios', icon: <FileText className="w-5 h-5" /> },
    { id: 'agenda', label: 'Agenda Legislativa', icon: <Scale className="w-5 h-5" /> },
    { id: 'legisladores', label: 'Legisladores', icon: <Users className="w-5 h-5" /> },
    { id: 'boveda', label: 'Bóveda', icon: <FolderOpen className="w-5 h-5" /> },
  ];
  const pageTitles = { dashboard: 'Dashboard', sesiones: 'Sesiones', oficios: 'Oficios Salientes', agenda: 'Agenda Legislativa', legisladores: 'Legisladores', boveda: 'Bóveda Documental' };
  const commonProps = { darkMode, addToast, onNavigate: navigateToEntity };

  const renderPage = () => {
    switch (currentPage) {
      case 'dashboard': return <Dashboard sessions={sessions} oficios={oficios} projects={projects} legislators={legislators} {...commonProps} />;
      case 'sesiones': return <SessionsModule sessions={sessions} oficios={oficios} {...commonProps} onSave={(d) => { if (d.id) setSessions(p => p.map(s => s.id === d.id ? { ...s, ...d } : s)); else setSessions(p => [...p, { ...d, id: Date.now(), activo: true }]); }} onDelete={(id) => setSessions(p => p.map(s => s.id === id ? { ...s, activo: false } : s))} />;
      case 'oficios': return <OficiosModule oficios={oficios} sessions={sessions} {...commonProps} onSave={(d) => { if (d.id) setOficios(p => p.map(o => o.id === d.id ? { ...o, ...d } : o)); else setOficios(p => [...p, { ...d, id: Date.now(), activo: true }]); }} onDelete={(id) => setOficios(p => p.map(o => o.id === id ? { ...o, activo: false } : o))} />;
      case 'agenda': return <AgendaModule projects={projects} commissions={commissions} legislators={legislators} {...commonProps} onSave={(d) => { const now = new Date().toISOString().split('T')[0]; if (d.id) setProjects(p => p.map(pr => pr.id === d.id ? { ...pr, ...d, fecha_actualizacion: d.fecha_actualizacion || now } : pr)); else setProjects(p => [...p, { ...d, id: Date.now(), activo: true, fecha_actualizacion: now }]); }} onDelete={(id) => setProjects(p => p.map(pr => pr.id === id ? { ...pr, activo: false } : pr))} />;
      case 'legisladores': return <LegislatorsModule legislators={legislators} commissions={commissions} {...commonProps} onSaveLegislator={(d) => { if (d.id) setLegislators(p => p.map(l => l.id === d.id ? { ...l, ...d } : l)); else setLegislators(p => [...p, { ...d, id: Date.now(), activo: true }]); }} onSaveCommission={(d) => { if (d.id) setCommissions(p => p.map(c => c.id === d.id ? { ...c, ...d } : c)); else setCommissions(p => [...p, { ...d, id: Date.now(), activo: true }]); }} onDeleteLegislator={(id) => setLegislators(p => p.map(l => l.id === id ? { ...l, activo: false } : l))} onDeleteCommission={(id) => setCommissions(p => p.map(c => c.id === id ? { ...c, activo: false } : c))} />;
      case 'boveda': return <VaultModule documents={documents} sessions={sessions} oficios={oficios} projects={projects} {...commonProps} onSaveDocument={(doc) => setDocuments(p => [...p, doc])} onDeleteDocument={(id) => setDocuments(p => p.map(d => d.id === id ? { ...d, activo: false } : d))} />;
      default: return <Dashboard sessions={sessions} oficios={oficios} projects={projects} legislators={legislators} {...commonProps} />;
    }
  };

  if (!config.setupComplete) return (
    <>
      <SetupWizard onComplete={setConfig} initialDark={config.darkMode} />
      <ToastContainer toasts={toasts} removeToast={removeToast} />
      <style>{`@keyframes slide-in{from{transform:translateX(100%);opacity:0}to{transform:translateX(0);opacity:1}}@keyframes scale-in{from{transform:scale(0.95);opacity:0}to{transform:scale(1);opacity:1}}::-webkit-scrollbar{width:6px}::-webkit-scrollbar-track{background:transparent}::-webkit-scrollbar-thumb{background:${darkMode ? '#374151' : '#d1d5db'};border-radius:3px}`}</style>
    </>
  );

  return (
    <div className={darkMode ? 'dark' : ''}>
      <div className={`min-h-screen flex ${darkMode ? 'bg-gray-950 text-white' : 'bg-gray-50 text-gray-900'}`}>
        <aside className={`fixed left-0 top-0 h-full z-40 transition-all duration-300 flex flex-col ${darkMode ? 'bg-gray-900 border-gray-800' : 'bg-white border-gray-200'} border-r ${sidebarOpen ? 'w-64' : 'w-16'}`}>
          <div className="flex items-center gap-3 px-4 py-4 border-b border-gray-800/50"><div className="w-9 h-9 rounded-xl bg-gradient-to-br from-indigo-500 to-purple-600 flex items-center justify-center flex-shrink-0"><Gavel className="w-5 h-5 text-white" /></div>{sidebarOpen && (<div className="min-w-0"><p className="text-sm font-bold truncate">Segundo Cerebro</p><p className={`text-[10px] truncate ${darkMode ? 'text-gray-500' : 'text-gray-400'}`}>Legislativo</p></div>)}</div>
          <nav className="flex-1 py-4 px-2 space-y-1 overflow-y-auto">{navItems.map(item => (<button key={item.id} onClick={() => setCurrentPage(item.id)} className={`w-full flex items-center gap-3 px-3 py-2.5 rounded-xl text-sm font-medium transition-all ${currentPage === item.id ? darkMode ? 'bg-indigo-500/10 text-indigo-400' : 'bg-indigo-50 text-indigo-600' : darkMode ? 'text-gray-400 hover:bg-gray-800 hover:text-white' : 'text-gray-600 hover:bg-gray-100 hover:text-gray-900'} ${!sidebarOpen ? 'justify-center px-2' : ''}`}>{item.icon}{sidebarOpen && <span className="truncate">{item.label}</span>}</button>))}</nav>
          {sidebarOpen && (<div className={`mx-3 mb-4 p-3 rounded-xl ${darkMode ? 'bg-gray-800' : 'bg-gray-50'}`}><p className={`text-xs font-medium ${darkMode ? 'text-gray-300' : 'text-gray-700'}`}>{config.nombre_secretario || 'Secretario'}</p><p className={`text-[10px] ${darkMode ? 'text-gray-500' : 'text-gray-400'}`}>{config.cedula || ''} · {config.periodo_sesiones}</p></div>)}
          <div className="p-2 border-t border-gray-800/50"><button onClick={() => setSidebarOpen(p => !p)} className={`w-full flex items-center gap-3 px-3 py-2.5 rounded-xl text-sm ${darkMode ? 'text-gray-400 hover:bg-gray-800' : 'text-gray-600 hover:bg-gray-100'} ${!sidebarOpen ? 'justify-center' : ''}`}>{sidebarOpen ? <ChevronLeft className="w-5 h-5" /> : <ChevronRight className="w-5 h-5" />}{sidebarOpen && <span>Colapsar</span>}</button></div>
        </aside>

        <div className={`flex-1 transition-all duration-300 ${sidebarOpen ? 'ml-64' : 'ml-16'}`}>
          <header className={`sticky top-0 z-30 flex items-center justify-between px-6 py-3 border-b backdrop-blur-xl ${darkMode ? 'bg-gray-950/80 border-gray-800' : 'bg-white/80 border-gray-200'}`}>

'text-gray-600' : 'text-gray-300'}`} />}<button onClick={() => c.page && setCurrentPage(c.page)} className={`text-sm ${i === arr.length - 1 ? 'font-semibold' : darkMode ? 'text-gray-400 hover:text-white' : 'text-gray-500 hover:text-gray-900'}`}>{c.label}</button></React.Fragment>))}
            </div>

              <button onClick={() => setShowCommandPalette(true)} className={`flex items-center gap-2 px-3 py-1.5 rounded-lg text-sm ${darkMode ? 'bg-gray-800 text-gray-400 hover:text-white' : 'bg-gray-100 text-gray-500 hover:text-gray-700'}`}><Search className="w-4 h-4" /><span className="hidden sm:inline">Buscar...</span><kbd className={`hidden sm:inline px-1.5 py-0.5 rounded text-[10px] ${darkMode ? export const generateQR = async (data) => {
  const payload = JSON.stringify(data);
  try {
    // Intenta usar Electron API, fallback a librería web si es necesario
    if (window.legisAPI) {
      return await window.legisAPI.qr.generate(payload);
    }
    // Fallback web (requiere instalar 'qrcode' también en react)
    // return await import('qrcode').then(q => q.toDataURL(payload));
  } catch (e) {
    console.error("QR Generation failed", e);
    return null;
  }
};

export const createVersionSnapshot = async (projectData, user) => {
  // Lógica para guardar versión en DB local via IPC
  const payload = {
    projectId: projectData.id,
    snapshot: JSON.stringify(projectData),
    author: user,
    timestamp: new Date().toISOString()
  };
  
  if (window.legisAPI) {
    return await window.legisAPI.db.query('versions', payload);
  }
  console.warn("Versioning: API not available (Web Mode?)");
};

🔧 Troubleshooting
Error
Solución
Cannot find module 'better-sqlite3'
Ejecutar npm rebuild better-sqlite3. Requiere Python y herramientas de compilación (Visual Studio Build Tools en Windows).
Pantalla en blanco en Electron
Verificar vite.config.js -> base: './' y que dist/index.html existe.
Git sync falla
Asegurar que el repo tiene un origin configurado y estás logueado en git (git config --global ...).
Telegram no responde
Verificar token en .env y que el bot no tenga otro webhook activo.
